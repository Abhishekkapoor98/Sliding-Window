void solve()
{
	int t;
	cin>>t;

	while(t--)
	{
		int n;
		cin>>n;

		vi arr(n);

		for(int i=0;i<n;i++)
			cin>>arr[i];

		int k;
		cin>>k;


		deque<int> look;

		int i=0,j=0;

		while(j < n)
		{
			if(look.empty() == true)
				look.PB(arr[j]);
			if(arr[j] < look[0])
			{
				while(look[0] < arr[j]){
					look.pop_front();
					if(look.empty() == true)
						break;
				}

				look.PB(arr[j]);
			}
			else
				look.push_front(arr[j]);

			if(j-i+1 < k)
				j++;
			else if(j-i+1 == k)
			{
				cout<<look[0]<<" ";
				if(look[0] == arr[i])
					look.pop_front();
				i++;
				j++;
			}

		}
	}
}
