void solve()
{
	int t;
	cin>>t;

	debug(t);
	while(t--)
	{
		string str,ptrn;
		cin>>str>>ptrn;

		map<char,int> look;

		for(int i=0;i<ptrn.length();i++)
			look[ptrn[i]]++;


		int ans = 0,i=0,j=0,n=str.length(),count = look.size(),k = ptrn.length();

		while(j<n)
		{
			if(look.find(str[j]) != look.end())
			{
				look[str[j]]--;
				if(look[str[j]] == 0)
					count--;
			}
			if(j-i+1 < k)
				j++;
			else if(j-i+1 == k)
			{
				if(count == 0)
					ans++;
				if(look.find(str[i]) != look.end())
				{
					look[str[i]]++;
					if(look[str[i]] == 1)
						count++;
				}
				i++;
				j++;
			}
		}
		cout<<ans<<endl;
	}	
}
